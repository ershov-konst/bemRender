bemRender

express.js view render based on BEM methodology

how to use

configuration

app.set('view engine', 'json');
app.engine('json', require('./lib/bemRender')({
express : app,
blocksPath : __dirname + "/blocks",
publicPath : __dirname + "/public"
}));

render

app.get('/hello', function(req, res){
res.render("index");
});
